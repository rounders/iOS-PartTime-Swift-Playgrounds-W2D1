//: ## Challenge 2


func pluralize(count: Int) {
    if count == 1 {
        print("There is \(count) bicycle")
    } else if (count == 0) {
        print("There are zero bicycles")
    } else {
        print("There are \(count) bicycles")
    }
}


//: ## Challenge 3


func pluralizedString(count: Int) -> String {
    if count == 1 {
        return "There is \(count) bicycle"
    } else if (count == 0) {
        return "There are zero bicycles"
    } else {
        return "There are \(count) bicycles"
    }
}



//: ## Bonus Challenge


func pluralizedWord(singular: String, pluralNoun: String, count: Int) -> String {
    if count == 1 {
        return "There is \(count) \(singular)"
    } else if (count == 0) {
        return "There are zero \(pluralNoun)"
    } else {
        return "There are \(count) \(pluralNoun)"
    }
}
